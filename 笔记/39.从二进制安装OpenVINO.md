# 从二进制安装OpenVINO

1.  wget https://apt.repos.intel.com/intel-gpg-keys/GPG-PUB-KEY-INTEL-SW-PRODUCTS.PUB
2.  sudo apt-key add GPG-PUB-KEY-INTEL-SW-PRODUCTS.PUB
3.  echo "deb https://apt.repos.intel.com/openvino ubuntu24 main" | sudo tee /etc/apt/sources.list.d/intel-openvino.list
4.  sudo apt update
5.  sudo apt install openvino-2025.3.0
